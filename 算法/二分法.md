## 代码

```cpp
#include<iostream>
#include<vector>
using namespace std;

int fun(const vector<int> &vec,const int &target)
{
	int left = 0, right = vec.size() - 1;
	int mid = 0;
	int res = -1;
	while (left <= right)
	{
		mid = (left + right) / 2;
		if (vec[mid] < target)
		{
			left = mid+1;
		}
		else if (vec[mid] > target)
		{
			right = mid-1;
		}
		else {
			res = mid;
			break;
		}			
	}
	return res;
}
void main()
{
	vector<int> vec = { 1,2,3,5,8 };
	cout << fun(vec, 6);
}
```
