void merge_sort(vector<int> &vec, int left, int right)
{
	if (left < right)
	{
		int mid = (left + right) / 2;

		merge_sort(vec, left, mid);
		merge_sort(vec, mid + 1, right);
		merge(vec, left, mid, right);
	}
}
