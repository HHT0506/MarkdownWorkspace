```cpp
#include<iostream>
#include<vector>
using namespace std;

void Sort(vector<int> &nums, int left, int right)
{
	if (right <= left)
		return;
	int _left = left, _right = right;//保存初始值

	int key = nums[left];
	bool r_start = true;
	while (left < right)
	{
		if (r_start)
		{
			while (nums[right] >= key)
			{
				right--;
				if (right == left)
					break;
			}
				
			nums[left] = nums[right];
			r_start = false;
		}
		else
		{
			while (nums[left] <= key)
			{
				left++;
				if (right == left)
					break;
			}
			nums[right] = nums[left];
			r_start = true;
		}
	}
	nums[right] = key;
	Sort(nums, _left, left - 1);
	Sort(nums, right + 1, _right);

}
void quickSort(vector<int> &nums)
{
	int left = 0, right = nums.size() - 1;
	Sort(nums, left, right);

}
int main()
{
	vector<int> nums{ 5,4,7,2,9,1,77};
	quickSort(nums);
	for (auto it : nums)
		cout << it<<" ";
	cout << endl;

	return 0;
}
```

while循环优化写法

```cpp
#include<iostream>
#include<vector>
using namespace std;


void Sort(vector<int> &nums, int left, int right)
{
	if (right <= left)
		return;
	int _left = left, _right = right;//保存初始值

	int key = nums[left];
	while (left < right)
	{
		while (nums[right] >= key&&left < right)
			right--;
		nums[left] = nums[right];
		while (nums[left] <= key&&left < right)
			left++;
		nums[right] = nums[left];
	}
	nums[right] = key;
	Sort(nums, _left, left - 1);
	Sort(nums, right + 1, _right);

}
void quickSort(vector<int> &nums)
{
	int left = 0, right = nums.size() - 1;
	Sort(nums, left, right);

}
int main()
{
	vector<int> nums{ 5,4,7,2,9,1,77,0 };
	quickSort(nums);
	for (auto it : nums)
		cout << it << " ";
	cout << endl;

	return 0;
}
```
