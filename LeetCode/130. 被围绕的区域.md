```cpp
class Solution {
public:
    void solve(vector<vector<char>>& board) {
        int row=board.size();
        int col=board[0].size();
        if(row<=2||col<=2)
            return ;

        for(int j=0;j<col;j++)
            if(board[0][j]=='O')
                bfs(board,0,j);
        for(int i=1;i<row;i++)
            if(board[i][col-1]=='O')
                bfs(board,i,col-1);
        for(int j=col-2;j>=0;j--)
            if(board[row-1][j]=='O')
                bfs(board,row-1,j);
        for(int i=row-2;i>=0;i--)
            if(board[i][0]=='O')
                bfs(board,i,0);
        for(int i=0;i<row;i++)
        {
            for(int j=0;j<col;j++)

                if(board[i][j]=='#')
                    board[i][j]='O';
                else if(board[i][j]=='O')
                    board[i][j]='X';
        }
    }
    void bfs(vector<vector<char>> &board,int i,int j)
    {
        int row=board.size();
        int col=board[0].size();
        queue<pair<int,int>> que;
        int pos[5]={-1,0,1,0,-1};
        que.push(pair<int,int>{i,j});
        board[i][j]='#';
        while(!que.empty())
        {
            pair<int,int> currentPos=que.front();
            que.pop();
            for(int m=0;m<4;m++)
            {
                int x=currentPos.first+pos[m];
                int y=currentPos.second+pos[m+1];
                if(x>=0&&x<row&&y>=0&&y<col&&board[x][y]=='O')
                {
                    que.push(pair<int,int> {x,y});
                    board[x][y]='#';
                }

            }
        }
    }
};
```
