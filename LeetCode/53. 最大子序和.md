## 暴力法代码

类似于冒泡排序，使用双层循环，遍历所有情况。

```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int size=nums.size();
        int max=nums[0];
        int sum=0;
        for(int i=0;i<size-1;i++){
            sum=nums[i];
            if(sum>max)
                max=sum;
            for(int j=i+1;j<size;j++)
            {
                sum+=nums[j];
                if(sum>max)
                    max=sum;
            }
        }
        if(nums[size-1]>max)
           max=nums[size-1];
        return max;
    }
};
```
